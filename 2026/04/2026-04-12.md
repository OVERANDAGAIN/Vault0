```
import os
import openai
from typing import List, Dict, Any

class LLMClient:
    """
    Unified LLM Client supporting multiple sources (vapi, litellm) 
    configured via environment variables.
    """
    def __init__(self):
        self.source = os.getenv("LLM_SOURCE", "vapi").lower()
        self.api_key = os.getenv("LLM_API_KEY")
        self.model_name = os.getenv("LLM_MODEL")
        
        # Default base URLs based on source
        default_urls = {
            "vapi": "https://api.gpt.ge/v1",
            "litellm": "https://litellm.mybigai.ac.cn"
        }
        
        self.base_url = os.getenv("LLM_BASE_URL", default_urls.get(self.source, default_urls["vapi"]))
        
        if not self.api_key:
            # We don't raise error immediately to allow fallback or manual config, 
            # but in practice it should be set.
            print(f"Warning: LLM_API_KEY is not set for source '{self.source}'")
        
        if not self.model_name:
            # Provide sensible defaults if not specified
            self.model_name = "claude-opus-4-5-20251101" if self.source == "vapi" else "gpt-4.1-mini-2025-04-14"

        self.client = openai.OpenAI(
            api_key=self.api_key,
            base_url=self.base_url
        )

    def create_completion(self, messages: List[Dict[str, str]], **kwargs) -> str:
        """
        Standard chat completion call.
        """
        model = kwargs.pop("model", self.model_name)
        temperature = kwargs.pop("temperature", 0.7)
        max_tokens = kwargs.pop("max_tokens", 1000)
        
        try:
            response = self.client.chat.completions.create(
                model=model,
                messages=messages,
                temperature=temperature,
                max_tokens=max_tokens,
                **kwargs
            )
            return response.choices[0].message.content
        except Exception as e:
            return f"Error calling LLM ({self.source}): {e}"

# Simple getter for a shared instance
_shared_client = None

def get_llm_client() -> LLMClient:
    global _shared_client
    if _shared_client is None:
        _shared_client = LLMClient()
    return _shared_client
```
