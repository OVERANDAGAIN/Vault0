```bash
RealPref preferences
    ├── user_profiles.json
    │       ├── preferences          原始细粒度偏好
    │       ├── interests            preferences.Topic 一级类别 + persona keyword
    │       └── intent_distribution  interests/preferences 经 mappings 映射后的任务意图 prior
    │
    ├── initial_queries.json
    │       ├── profile_based_queries     根据 intent_distribution 从 Infinite-Chats 采样
    │       ├── preference_queries        根据 preferences[:2] 生成
    │       └── random_queries            从 Infinite-Chats 全局随机采样
    │
    └── topic_pool.json
            ├── Infinite-Chats categories
            └── RealPref preference topics / sample queries
```
